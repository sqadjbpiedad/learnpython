# Write solution here
class Livestock:
    def __init__(self, species, count):
        self.species = species
        self.count = count

    def herd_status(self):
        if self.count > 0:
            return "Herd registered"
        return "No livestock"

    def __str__(self):
        return f"{self.species}: {self.count} head(s) ({self.herd_status()})"

class Farmer:
    def __init__(self, name):
        self.name = name
        self.inventory = []

    def add_livestock(self, livestock):
        self.inventory.append(livestock)

    def remove_livestock(self, species):
        self.inventory = [
            animal for animal in self.inventory
            if animal.species != species
        ]

    def __str__(self):
        if self.inventory:
            herd_info = "\n".join(str(animal) for animal in self.inventory)
        else:
            herd_info = "No registered livestock."
            
        return f"Farmer: {self.name}\nRegistered Herd:\n{herd_info}"
