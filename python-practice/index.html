class Farm:
    def __init__(self, name, crop, hectares):
        self.name = name
        self.crop = crop
        self.hectares = hectares

    def estimate_production(self):
        """Assume standard yield of 4.5 tons/ha"""
        return self.hectares * 4.5

    # OVERRIDING the standard print function
    def __str__(self):
        return f"FARM: {self.name} | CROP: {self.crop} | AREA: {self.hectares} ha"

class Livestock:
    """
    A class to represent livestock.

    Attributes:
        species (str): The species of the livestock.
        count (int): The number of animals.
    """

    def __init__(self, species: str, count: int):
        """
        Initialize a Livestock object.

        Args:
            species (str): The species of the livestock.
            count (int): The number of animals.
        """
        self.species = species
        self.count = count

    def check_status(self) -> str:
        """
        Determine the status of the herd based on its size.

        Returns:
            str: "Small Herd" if count < 5, otherwise "Commercial Herd".
        """
        if self.count < 20:
            return "Small Herd"
        else:
            return "Commercial Herd"
